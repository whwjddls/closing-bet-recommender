**Goal:** 서브시스템 4 API(`/recommendations/{date}`·`/stock/{code}`·`/performance`·`/universe`·`/health`)에 바인딩하여 추천보드·종목상세·성과추적 3화면과 7개 컴포넌트를 TDD로 구현한다. 10분 결정창 UX(top3 강조·알림·정렬/필터), 빈/저레짐 배너, provisional 워터마크, 콜드스타트 게이팅, RISK_OFF 시 스캐너 전면 유지를 vitest+RTL 테스트로 고정한다.

**Architecture:** React 18 + Vite + TypeScript SPA. `src/api/client.ts`가 5개 엔드포인트와 응답 타입(서브시스템 4 SQLite 스키마 필드명에 1:1 고정)을 캡슐화한다. 페이지(`pages/`)는 순수 표현 컴포넌트(`components/`)를 조립하고, 도메인 헬퍼(`lib/`: 등급 컷오프·배지·포맷·알림)는 순수함수로 분리해 단위 테스트한다. 모든 외부 호출은 `fetch` 목으로만 테스트하며 네트워크 호출은 금지한다.

**Tech Stack:** React 18, react-router-dom 6, Vite 5, TypeScript 5, vitest + jsdom, @testing-library/react/user-event/jest-dom, lightweight-charts(StockDetail 일봉; 테스트는 `vi.mock`). 미니차트는 의존성 없는 SVG 스파크라인.

```ts
  ticker: string;
  name: string;
  market: Market;
  rank: number;
  price_provisional: number;
  buy_price_provisional: number;
  buy_price_final: number | null;
  s_shin: number;
  s_geo: number;
  rvol_confirm: number;

  regime_mult: number;
  veto: number;
  core: number;
  final: number;
  grade: Grade;
  near_252: number | null;
  near_60: number | null;
  rvol: number | null;
  target_price: number;
  stop_price: number;
  provisional_flag: boolean;
  base_flag: boolean;
  spark: number[];
}

export interface RegimeState {
  market: Market;
  index_level: number;
  ma5: number;
  ma5_prev: number;
  cond_a: boolean;
  cond_b: boolean;
  regime_mult: number;
}

export interface RecommendationsResponse {
  run_date: string;
  board_published: boolean;
  coverage_pct: number;
  session_type: string;
  regimes: RegimeState[];
  recommendations: Recommendation[];
}

export interface Candle {
  date: string;
  open: number;
  high: number;
  low: number;
  close: number;
  volume: number;
}

export interface BaseBox {
  top: number;
  bottom: number;
  from: string;
  to: string;
}

export interface StockDetailResponse {
  ticker: string;
  name: string;
  market: Market;
  price_provisional: number;
  grade: Grade;
  core: number;
  final: number;
  s_shin: number;
  rvol_confirm: number;
  supply_tilt: number;
  regime_mult: number;
  veto: number;
  candles: Candle[];
  high_52w: number;
  prior_high: number;
  base_box: BaseBox | null;
  provisional_flag: boolean;
}

export interface PerfRow {
  ticker: string;
  name: string;
  grade: Grade;
  buy_price_final: number | null;
  vwap_0900_1000: number | null;
  morning_return: number | null;
  outcome: Outcome;
  dart_overnight_flag: boolean;
}

export interface PerfAggregate {
  hit_rate: number | null;
  avg_morning_return: number | null;
  sample_size: number;
  cold_start: boolean;
  cumulative_curve: { date: string; cum_return: number }[];
  by_grade: { grade: Grade; hit_rate: number; n: number }[];
  by_regime: { regime: string; hit_rate: number; n: number }[];
}

export interface PerformanceResponse {
  eval_date: string;
  aggregate: PerfAggregate;
  rows: PerfRow[];
}

export interface UniverseRow {
  ticker: string;
  name: string;
  market: Market;
  sec_type: string;
  avg_value_20d: number;
  is_managed: boolean;
  is_warning: boolean;
  is_caution: boolean;
  eligible: boolean;
}

export interface UniverseResponse {
  as_of: string;
  rows: UniverseRow[];

export interface HealthResponse {
  status: 'OK' | 'DEGRADED' | 'DOWN';
  kis_coverage_pct: number | null;
  board_published: boolean;
  last_run_date: string | null;
  reason: string | null;
}

const BASE_URL =
  (import.meta.env?.VITE_API_BASE as string | undefined) ?? 'http://localhost:8000';

async function getJson<T>(path: string): Promise<T> {
  const res = await fetch(`${BASE_URL}${path}`);
  if (!res.ok) {
    throw new Error(`API ${path} failed: ${res.status}`);
  }
  return (await res.json()) as T;
}

export const fetchRecommendations = (date: string) =>
  getJson<RecommendationsResponse>(`/recommendations/${date}`);
export const fetchStock = (code: string) =>
  getJson<StockDetailResponse>(`/stock/${code}`);
export const fetchPerformance = () => getJson<PerformanceResponse>(`/performance`);
export const fetchUniverse = () => getJson<UniverseResponse>(`/universe`);
export const fetchHealth = () => getJson<HealthResponse>(`/health`);
```

```ts
export function gradeFromCore(core: number): Grade {
  if (core >= 0.8) return 'S';
  if (core >= 0.6) return 'A';
  if (core >= 0.4) return 'B';
  return 'C';
}
```

```ts
export function formatPercent(
  ratio: number | null | undefined,
```

```tsx
const W = 80;
const H = 24;

export default function MiniChart({ data }: { data: number[] }) {
  if (data.length < 2) {
    return <svg data-testid="mini-chart" data-empty="true" width={W} height={H} />;
  }
  const min = Math.min(...data);
  const max = Math.max(...data);
  const span = max - min || 1;
  const points = data
    .map((v, i) => {
      const x = (i / (data.length - 1)) * W;
      const y = H - ((v - min) / span) * H;
      return `${x.toFixed(1)},${y.toFixed(1)}`;
    })
    .join(' ');
  const trend = data[data.length - 1] >= data[0] ? 'up' : 'down';
  return (
    <svg data-testid="mini-chart" data-trend={trend} width={W} height={H} className={`mini mini-${trend}`}>
      <polyline points={points} fill="none" strokeWidth={1.5} />
    </svg>
  );
}
```

```ts
import type { Recommendation } from '../api/client';

export function notifyTop3(recs: Recommendation[]): void {
  if (typeof Notification === 'undefined') return;
  if (Notification.permission !== 'granted') return;
  const top3 = recs.filter((r) => r.rank <= 3).map((r) => r.name);
  if (top3.length === 0) return;
  new Notification('종가베팅 top3 추천', { body: top3.join(', ') });
}
```

```tsx
import { useEffect, useState } from 'react';
import { fetchPerformance, type PerformanceResponse } from '../api/client';
import { formatPercent } from '../lib/format';
import PerfTable from '../components/PerfTable';
import MiniChart from '../components/MiniChart';

function pct(ratio: number | null): string {
  return ratio === null ? 'N/A' : `${Math.round(ratio * 100)}%`;
}

export default function Performance() {
  const [data, setData] = useState<PerformanceResponse | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    fetchPerformance().then(setData).catch((e) => setError(String(e)));
  }, []);

  if (error) return <p data-testid="perf-error">성과를 불러오지 못했습니다: {error}</p>;
  if (!data) return <p>로딩 중…</p>;

  const a = data.aggregate;

  return (
    <main>
      <h1>성과 추적 ({data.eval_date})</h1>

      <section data-testid="agg-panel" data-cold-start={a.cold_start} className={a.cold_start ? 'agg cold-start' : 'agg'}>
        {a.cold_start && (
          <p data-testid="cold-start-caption" className="cold-start-caption">
            데이터 누적 중 (표본 {a.sample_size} &lt; 30) — 적중률은 참고용입니다.
          </p>
        )}
        <div className="agg-grid">
          <span data-testid="agg-hit-rate">적중률 {pct(a.hit_rate)} (n={a.sample_size})</span>
          <span>평균 오전수익률 {formatPercent(a.avg_morning_return)}</span>
        </div>

        <div data-testid="cum-curve">
          누적곡선 <MiniChart data={a.cumulative_curve.map((c) => c.cum_return)} />
        </div>

        <div className="by-grade">
          {a.by_grade.map((g) => (
            <span key={g.grade} data-testid={`by-grade-${g.grade}`}>
              {g.grade} {pct(g.hit_rate)} (n={g.n})
            </span>
          ))}
        </div>

        <div className="by-regime">
          {a.by_regime.map((r) => (
            <span key={r.regime} data-testid={`by-regime-${r.regime}`}>
              레짐 {r.regime} {pct(r.hit_rate)} (n={r.n})
            </span>
          ))}
        </div>
      </section>

      <h2>어제 픽 (확정 채점)</h2>
      <PerfTable rows={data.rows} />
    </main>
  );
}
```
